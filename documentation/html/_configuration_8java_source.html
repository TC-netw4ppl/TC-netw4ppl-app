122  public void createHashMaps(Context context, JSONObject config_content) {
123  // récupère toutes les clés de l'objet json config_content
124  Iterator<String> iterator_table = config_content.keys();
125  while (iterator_table.hasNext()) {
126  String key_table = iterator_table.next();
127  try {
128  JSONObject table = config_content.getJSONObject(key_table);
129  // récupère toutes les clés de l'objet iterator_table
130  Iterator<String> iterator_elements = table.keys();
131 
132  if (!key_table.equals("fields")) {
133  // create the ArrayList
134  ArrayList<DataElement> data_array = new ArrayList<DataElement>();
135  // create a new empty HashMap (like a database table)
136  HashMap<String, DataElement> data_table = new HashMap<>();
137  data_array.add(new DataElement());
138  while (iterator_elements.hasNext()) {
139  String key_element = iterator_elements.next();
140  JSONObject element = table.getJSONObject(key_element);
141  DataElement data_elem = new DataElement(key_element, element.toString());
142 
143  // add the element to the array
144  data_array.add(data_elem);
145 
146  // add the element to the hashmap. it's like an insert into a table in a database.
147  data_table.put(key_element, data_elem);
148  }
149  // then add everything to the more global HashMap
150  hashMap_database.put(key_table, data_array);
151  hashMap_datatables.put(key_table, data_table);
152 
153  // create the adapters from the array just created
154  ArrayAdapter adapter = new ArrayAdapter(context, android.R.layout.simple_spinner_item, data_array);
155  adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
156  hashMap_adapters.put(key_table, adapter);
157  }
158  else {
159  createArrayFields(table);
160  }
161  } catch (JSONException e) {
162  e.printStackTrace();
